import axios from 'axios'

export function Contributors() {
  axios.get(`https://api.github.com/repos/HUAT-FSAC/Guidance-Astro/contributors`)
    .then((res) => {
      const users = res.data;
      const detail = users.map((user) => {
        console.log(user.login);
        return (<image src={user.avatar_url} />)
        // document.getElementById("asd").innerHTML = user.login;
      });
      // console.log(detail);
      
      return detail;
    }).catch((e) => console.log(e));
    return "";
}
